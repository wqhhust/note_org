** startup configuration for iTerms

  - name: sample-two-panes
    root: ~/Code/sample/www
    layout: even-horizontal
    panes:
      - ls -l
      - ptipython
      - mycli mysql://_read:_read@10....:3306/redb
      - python manage.py runserver 10.249.66.124:5555

#+END_SRC
